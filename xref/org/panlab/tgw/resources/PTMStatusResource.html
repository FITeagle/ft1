35  // The Java class will be hosted at the URI path "/dummy"
36  @Path("/")
37  public class PTMStatusResource 
38  {
39      static int classIndex=0;
40      int j=0;
41  
42      public PTMStatusResource()
43      {
44          j = classIndex++;
45      }
46  
47      public static String createError(String errorInfo, String ra_id, String type)
48      {
49          return XMLUtil.xmlHeader
50             +"<"
51             +type+" status=\"FAIL\" code=\"410\"><uuid type=\"string\">"
52             +ra_id+"</uuid><info>"
53             +errorInfo+"</info></"
54             +type+">";
55      }
56      public static String createOK(String ra_id, String type)
57      {
58          return XMLUtil.xmlHeader
59             +"<"
60             +type+" status=\"SUCCESS\">"+
61             (ra_id!=null?"<uuid type=\"string\">"+ra_id+"</uuid>":"")+
62             "</"+type+">";
63      }
64      private String createOK(String ra_id, String type, String request_id)
65      {
66          return XMLUtil.xmlHeader
67             +"<"
68             +type+" status=\"SUCCESS\"  requestId=\""+request_id+"\">"+
69             (ra_id!=null?"<uuid type=\"string\">"+ra_id+"</uuid>":"")+
70             "</"+type+">";
71      }
72      // The Java method will process HTTP GET requests
73      @GET
74      @Produces("text/xml")
75      //@Consumes("text/xml")
76      public String getResource( ) 
77      {
78      	List<? extends Ptm> ptms = RepoAdapter.getPtms();
79          String temp = XMLUtil.xmlHeader;
80          temp+="<ptms>";
81          
82          for (Ptm ptm : ptms)
83          {
84              String tmp = ptm.getCommonName();
85              temp += "<"+tmp+">"+org.panlab.tgw.App.statusText(App.ptm_status.get(tmp))+"</"+tmp+">"+"\n";
86          }
87          temp+="</ptms>";
88          return temp;
89      }
