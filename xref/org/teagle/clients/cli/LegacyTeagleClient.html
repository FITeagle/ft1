77  	public void execVctCommand(final String username, final String vctname,
78  			final String[] result, final String command) {
79  		try {
80  			final URLConnection conn = this.reqProcUrl.openConnection();
81  			conn.setDoOutput(true);
82  			conn.setRequestProperty("Content-Type", "text/plain");
83  
84  			final String req = this.genRequest(username, vctname, command);
85  			conn.getOutputStream().write(req.getBytes());
86  			result[0] = Util.readStream(conn.getInputStream());
87  			Util.debug("Answer from reqproc: " + result[0]);
88  		} catch (final IOException e) {
89  			e.printStackTrace();
90  			throw new RuntimeException(e);
91  		}
92  	}
