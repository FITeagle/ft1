40  	String combine(final String[] s, final String glue) {
41  		final int k = s.length;
42  		if (k == 0)
43  			return "";
44  		final StringBuilder out = new StringBuilder();
45  		out.append(s[0]);
46  		for (int x = 1; x < k; ++x)
47  			out.append(glue).append(s[x]);
48  		return out.toString();
49  	}

91  	private String dropLog(final LogBook book, String log, final String indent) {
92  		for (int i = 0; i < book.entries.length; ++i) {
93  			final Object entry = book.entries[i];
94  			if (entry instanceof LogBook) {
95  				final LogBook inner = (LogBook) entry;
96  				log += "\nBEGIN INNER LOG: name=" + inner.name + " component="
97  						+ inner.component + "\n";
98  				log += this.dropLog(inner, log, indent + "    ");
99  				log += "\nEND INNER LOG: name=" + inner.name + " component="
100 						+ inner.component + "\n";
101 			} else {
102 				log += indent + entry;
103 				if (i < (book.entries.length - 1))
104 					log += "\n";
105 			}
106 		}
107 		return log;
108 	}
