105 	public void removeResourceInstanceWidget(final ResourceInstanceWidget widget) {
106 		for (final Control control : this.content.getChildren())
107 			if ((control instanceof ResourceInstanceWidget)
108 					&& (control == widget)) {
109 				for (final Iterator<ConnectionView> it = this.connections
110 						.iterator(); it.hasNext();) {
111 					final ConnectionView connection = it.next();
112 					if ((connection.getSourceWidget() == widget)
113 							|| (connection.getTargetWidget() == widget))
114 						it.remove();
115 				}
116 				widget.dispose();
117 				this.updateContent();
118 			}
119 	}
