264 	public void handleEvent(final Event event) {
265 		switch (event.type) {
266 		case SWT.MouseDoubleClick:
267 			this.fireEditConfig(this.vctView);
268 			this.posOrig = null;
269 			break;
270 		case SWT.MouseDown:
271 			if ((event.widget != null) && (event.widget instanceof CLabel)) {
272 				final ConnectionView connection = new ConnectionView(
273 						event.widget == this.srcPin ? "CONTAINS" : "REFERENCES");
274 				connection.setSource(this, (CLabel) event.widget);
275 				connection.setDragPosition(((CLabel) event.widget).toDisplay(
276 						event.x, event.y));
277 				this.vctView.dragConnectionBegin(connection);
278 
279 			} else {
280 				this.posOrig = this.getLocation();
281 				this.posDragged = this.toDisplay(event.x, event.y);
282 			}
283 			break;
284 		case SWT.MouseMove:
285 			if ((event.widget != null) && (event.widget instanceof CLabel))
286 				this.vctView.dragConnectionUpdate(((CLabel) event.widget)
287 						.toDisplay(event.x, event.y));
288 			else if (this.posOrig != null) {
289 				final Point posNew = this.toDisplay(event.x, event.y);
290 				int x = (this.posOrig.x + posNew.x) - this.posDragged.x;
291 				int y = (this.posOrig.y + posNew.y) - this.posDragged.y;
292 
293 				if (x < 0)
294 					x = 0;
295 				if (y < 0)
296 					y = 0;
297 
298 				this.setLocation(x, y);
299 
300 				this.getParent().redraw();
301 			}
302 			break;
303 		case SWT.MouseUp:
304 			if ((event.widget != null) && (event.widget instanceof CLabel)) {
305 				final ResourceInstanceWidget target = this.vctView
306 						.dragConnectionEnd(((CLabel) event.widget).toDisplay(
307 								event.x, event.y), this, (CLabel) event.widget);
308 				if (target != null) {
309 					final ConnectionEvent connectionEvent = new ConnectionEvent();
310 					connectionEvent.sourceLabel = (CLabel) event.widget;
311 					connectionEvent.source = this;
312 					connectionEvent.target = target;
313 					connectionEvent.parent = this.vctView;
314 					this.fireConnectionNewEvent(connectionEvent);
315 				}
316 			} else if (this.posOrig != null) {
317 				this.posOrig = null;
318 				this.fireMoved(this.getLocation());
319 			}
320 			break;
321 		default:
322 			break;
323 		}
324 	}
