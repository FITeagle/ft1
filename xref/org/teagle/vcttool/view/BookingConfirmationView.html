98  	private String renderConfirmation(final Vct currentVct) throws IOException {
99  		final Template t_summary = new Template("templates/confirmation.html");
100 		final Template t_resource = new Template(
101 				"templates/confirmation-resource.html");
102 		final Template t_testbed = new Template(
103 				"templates/confirmation-testbed.html");
104 
105 		// TODO refetch vct
106 		// ModelManager.getInstance().refetchVct(currentVct);
107 
108 		final Map<String, Integer> testbedCounts = new HashMap<String, Integer>();
109 
110 		final double total_price = 0;
111 		int n = 0;
112 		String res = "";
113 
114 		for (final ResourceInstance instance : currentVct
115 				.getResourceInstances()) {
116 
117 			String provider = "";
118 			for (final Organisation org : ModelManager.getInstance()
119 					.listOrganisations())
120 				if (org.getResourceSpecs().contains(instance.getResourceSpec())) {
121 					provider = org.getName();
122 					break;
123 				}
124 			final Integer count = testbedCounts.get(provider);
125 			testbedCounts.put(provider, 1 + (count == null ? 0 : count));
126 
127 			res += t_resource.render("n", String.valueOf(++n), "name",
128 					instance.getCommonName(), "testbed", provider,
129 					"description", instance.getDescription(), "price", "n/a");
130 		}
131 
132 		n = 0;
133 		String testbeds = "";
134 		for (final String testbed : testbedCounts.keySet())
135 			testbeds += t_testbed.render("n", String.valueOf(++n), "name",
136 					testbed, "count",
137 					String.valueOf(testbedCounts.get(testbed)), "price", "n/a");
138 
139 		return t_summary.render("resources", res, "total_price",
140 				String.valueOf(total_price), "testbeds", testbeds);
141 	}
