142 		final HashMap<String, String> nameMapping = new HashMap<String, String>();
143 		final HashMap<ResourceInstance, ResourceInstance> instanceMapping = new HashMap<ResourceInstance, ResourceInstance>();
144 
145 		for (final ResourceInstance instance : data.getResourceInstances()) {
146 			final ResourceInstance newInstance = ModelManager.getInstance()
147 					.createResourceInstance(instance.getResourceSpec());
148 			for (final Configuration config : instance.getConfigurations())
149 				for (final Configuration newConfig : newInstance

151 					if (config.getCommonName()
152 							.equals(newConfig.getCommonName()))
153 						newConfig.setValue(config.getValue());
154 
155 			newInstance.setCommonName(instance.getResourceSpec()
156 					.getCommonName() + "-" + RandomKey.randomKey());
157 			nameMapping.put(instance.getCommonName(),
158 					newInstance.getCommonName());
159 
160 			instanceMapping.put(instance, newInstance);
161 
162 			// ResourceInstanceWidget widget = controller.getView(newVct);
163 			// widget.setLocation(position);
164 			newInstance.setGeometry(ModelManager.getInstance().createGeometry(
165 					instance.getGeometry().getX(),
166 					instance.getGeometry().getY()));
167 			newVct.addResourceInstance(newInstance);
168 
169 		}
170 
171 		for (final ResourceInstance newInstance : newVct.getResourceInstances()) {
172 			System.out.println("Fixing config for: "
173 					+ newInstance.getCommonName());
174 			for (final Configuration newConfig : newInstance

188 				} else
