48  	private void initMouseListener() {
49  		this.tree.addMouseListener(new MouseAdapter() {
50  			@Override
51  			public void mouseDoubleClick(final MouseEvent event) {
52  				final TreeItem item = VctSelectionController.this.tree
53  						.getItem(new Point(event.x, event.y));
54  				if (item != null) {
55  					final VctController vctController = (VctController) item
56  							.getData();
57  					VctSelectionController.this.root.getView().addVctView(
58  							vctController
59  									.getView(VctSelectionController.this.root
60  											.getView().getVctPane()),
61  							vctController,
62  							vctController.getVct().getCommonName());
63  				}
64  			}
65  		});
66  	}

68  	public void initViews() {
69  		final LegacyTeagleClient client = new LegacyTeagleClient(this.root.getConfig()
70  				.getUsername(), this.root.getConfig().getPassword(), this.root
71  				.getConfig().getReqprocUrl(), this.root.getConfig()
72  				.getRepoUrl());
73  		final List<Vct> vcts = client.getVCTs();
74  
75  		for (final Vct vct : vcts) {
76  			final TreeItem vctItem = new TreeItem(this.tree, SWT.NONE);
77  			vctItem.setText(vct.getCommonName());
78  			final VctController controller = new VctController(vct);
79  			this.controllers.put(vct.getCommonName(), controller);
80  
81  			vctItem.setData(controller);
82  
83  			this.tree.showItem(vctItem);
84  		}
85  
86  	}

102 	public VctController createEmptyVct(final String username, String vctName) {
103 		final Vct vct = ModelManager.getInstance().createVct();
104 		int i = 1;
105 		if (vctName == null)
106 			vctName = "Vct" + i;
107 		while (this.controllers.containsKey(vctName))
108 			vctName = "Vct" + i++;
109 
110 		vct.setCommonName(vctName);
111 		Person person = ModelManager.getInstance().findPersonByUserName(
112 				username);
113 		if (person == null) {
114 			person = ModelManager.getInstance().createPerson();
115 			person.setUserName(username);
116 			person.setFullName(username);
117 		}
118 		vct.setPerson(person);
119 		vct.setState(VctState.State.NEW);
120 		final VctController controller = new VctController(vct);
121 		this.controllers.put(vctName, controller);
122 		return controller;
123 	}
