360 	private String regexFind(String regex, String text) {
361 		Pattern pattern = Pattern.compile(regex, Pattern.DOTALL);
362 		Matcher matcher = pattern.matcher(text);
363 		return matcher.matches() ? matcher.group(1) : null;
364 	}

366 	// First OE operation ... no "uninteractive" version exists so we have to scrape the html
367 	// for the result afterwards. 
368 	private String call_putVCTSpec(HttpClient client, String oe_url, String vctid, String booking) 
369 			throws HttpException, IOException, ServletException {
370 		Util.info("call_putVCTSpec vct=" + vctid + " oe_url: " + oe_url);
371 		Util.info(booking);
372 		PostMethod post = new PostMethod(oe_url);
373 		
374 		HttpMethodRetryHandler rh = new HttpMethodRetryHandler() {
375 			
376 			@Override
377 			public boolean retryMethod(HttpMethod arg0, IOException arg1, int arg2) {
378 				Util.error("Exception during method call: " + arg1);
379 				return false;
380 			}
381 		};
382 		
383 		post.getParams().setParameter(HttpMethodParams.RETRY_HANDLER, rh);
384 		
385 		try {
386 			Part[] parts = {
387 				new StringPart("step",		"xdocmd"),
388 				new StringPart("op",		"putVCTSpec"),
389 				new StringPart("siteid",	"teagle"),
390 				new StringPart("appid",		"teagle.repository"),
391 				new StringPart("serviceid",	"VCTRegistry"),
392 				new StringPart("clientid",	""),
393 				new StringPart("v_vctid",	vctid),
394 				new FilePart("v_vctfile",	new ByteArrayPartSource("booking.xml", booking.getBytes())),	
395 				new StringPart("options",	"d0"),
396 				new StringPart("userId",	""),
397 			};
398 
399 			post.setRequestEntity(new MultipartRequestEntity(parts, post.getParams()));
400 			
401 			client.executeMethod(post);
402 			if (post.getStatusCode() != 200)
403 				throw new IOException("Unexpected response from OE: " + post.getStatusLine());
404 
405 			String response = Util.readStream(post.getResponseBodyAsStream());
406 			String result = regexFind(resultPattern, response);
407 			if (result==null)
408 				throw new ServletException("Could not find result tag, response=" + response);
409 			
410 			Util.info("vct=" + vctid + " result=" + result);
411 			return result;
412 		} finally {
413 			post.releaseConnection();
414 		}
415 	}

527 	/**
528 	 * Call the three OE operations that trigger a booking, and return an error if any of
529 	 * them fails.  
530 	 * @param async whether to book asynchronously or not 
531 	 * @param vctid the vct id, currently it's generated by concatenating the user and vct name
532 	 * @param booking a string that contains an xml request which will get sent to the OE
533 	 * @return an OrchestrateReturn instance that is parsed from the xml produced by the OE
534 	 */
535 	private OrchestrateReturn doBookViaOE(boolean async, String vctid, String booking) 
536 			throws IOException, ServletException {
537 		Util.info("doBookViaOE vctid=" + vctid);
538 		Util.info(booking);
539 		
540 		String oe_url    = properties.getProperty("oe.url");
541 		String oe_xml_url= properties.getProperty("oe.xml_url");
542 		
543 		if (oe_url==null || oe_xml_url==null)
544 			throw new ServletException("missing configuration item from reqprocessor.properties");
545 		
546 		HttpClient httpClient = OEHttpFactory.makeHttpClient(properties);
547 		
548 		String putVCTSpec_result = call_putVCTSpec(httpClient, oe_url, vctid, booking);
549 		if (! putVCTSpec_result.equalsIgnoreCase("OK"))
550 			throw new ServletException("Unexpected putVCTspec result: " + putVCTSpec_result);
551 
552 		//try { Thread.sleep(500); } catch (Exception e) { }
553 		
554 		String deployVCT_result = call_deployVCT(httpClient, oe_url, vctid);
555 		if (! deployVCT_result.equalsIgnoreCase("OK"))
556 		{
557 			String logUrl = regexFind(logPattern, deployVCT_result);
558 			if (logUrl != null)
559 			{
560 				OrchestrateReturn r = new OrchestrateReturn(1, "Error in orchestration");
561 				r.log = logUrl;
562 				return r;
563 			}
564 			throw new ServletException("Unexpected deployVCT result: " + deployVCT_result);
565 		}
566 
567 		//try { Thread.sleep(500); } catch (Exception e) { }
568 		
569 		String oeRet = async ?
570 			call_orchestrateAsync(httpClient, oe_xml_url, vctid) :
571 			call_orchestrateSync(httpClient, oe_xml_url, vctid);
572 		
573 		Util.debug("answer from OE: " + oeRet);
574 		try
575 		{
576 			return (OrchestrateReturn)xs.fromXML(oeRet);
577 		}
