123 	/**
124 	 * Returns a newly instantiated object of type clazz (all its fields are
125 	 * instantiated recursively as well).
126 	 */
127 	public static Object doDeepInstantiate(final Class<?> clazz)
128 			throws ReflectionException {
129 		try {
130 			final Object value = clazz.newInstance();
131 			for (final Field f : clazz.getFields())
132 				if (!f.getType().isPrimitive()
133 						&& !Modifier.isStatic(f.getModifiers())
134 						&& (f.get(value) == null))
135 					f.set(value, Util.doDeepInstantiate(f.getType()));
136 
137 			return value;
138 		} catch (final InstantiationException e) {
139 			throw new ReflectionException(e);
140 		} catch (final IllegalAccessException e) {
141 			throw new ReflectionException(e);
142 		}
143 	}

176 	public static Point[] bezierCubic(final Point p0, final Point p3,
177 			final int step) {
178 		final Point p1 = new Point(p0.x + 100, p0.y);
179 		final Point p2 = new Point(p3.x - 100, p3.y);
180 		if (p0.y == p3.y) {
181 			p1.y -= 80;
182 			p2.y -= 80;
183 		}
184 		final Point[] p = { p0, p1, p2, p3 };
185 
186 		final int mdist = Math.abs(p0.x - p3.x) + Math.abs(p0.y - p3.y);
187 		final int pts = mdist / step;
188 		final Point[] curve = new Point[pts + 1];
189 
190 		for (int k = 0; k <= pts; k++) {
191 			final double t = ((double) k) / pts;
192 			final double t1 = 1.0 - t;
193 			final double[] c = { t1 * t1 * t1, 3 * t1 * t1 * t, 3 * t1 * t * t,
194 					t * t * t };
195 			final double x = (c[0] * p[0].x) + (c[1] * p[1].x)
196 					+ (c[2] * p[2].x) + (c[3] * p[3].x);
197 			final double y = (c[0] * p[0].y) + (c[1] * p[1].y)
198 					+ (c[2] * p[2].y) + (c[3] * p[3].y);
199 			curve[k] = new Point((int) x, (int) y);
200 		}
201 
202 		return curve;
203 	}

253 	private static Map<String, Integer> logPrefixes = new HashMap<String, Integer>();
254 
255 	static {
256 		Util.logPrefixes.put("default", Util.LOG_WARN);
257 
258 		try {
259 			final Properties tmp = new Properties();
260 			final InputStream is = Util.resourceStream("log.properties");
261 			if (is == null)
262 				throw new IOException("can't find resource");
263 			tmp.load(is);
264 
265 			for (final Map.Entry<Object, Object> entry : tmp.entrySet()) {
266 				final String prefix = (String) entry.getKey();
267 				final String levelStr = ((String) entry.getValue())
268 						.toUpperCase();
269 				int level = -1;
270 
271 				for (int i = 0; i < Util.logLevels.length; i++)
272 					if (levelStr.equals(Util.logLevels[i]))
273 						level = i;
274 
275 				if (level < 0) {
276 					System.out.println("Unknown log level: " + levelStr);
277 					level = Util.LOG_WARN;
278 				}
279 
280 				Util.logPrefixes.put(prefix, level);
281 			}
282 		} catch (final IOException e) {
283 			System.out.println("Util: can't read log.properties resource");
284 		}
285 	}

290 	public static void log(final int level, final String txt) {
291 		try {
292 			StackTraceElement e = null;
293 			boolean foundUtil = false;
294 			final StackTraceElement[] trace = Thread.currentThread()
295 					.getStackTrace();
296 
297 			for (int i = 0; (e == null) && (i < trace.length); i++) {
298 				final String simpleName = Class
299 						.forName(trace[i].getClassName()).getSimpleName();
300 				final boolean equalsUtil = simpleName.equals("Util");
301 				foundUtil |= equalsUtil;
302 				if (foundUtil && !equalsUtil)
303 					e = trace[i];
304 			}
305 
306 			if (e == null)
307 				return;
308 
309 			final String clazz = Class.forName(e.getClassName())
310 					.getSimpleName();
311 			final String method = e.getMethodName();
312 			final String location = clazz + "." + method;
313 
314 			String longestPrefix = "";
315 			for (final String prefix : Util.logPrefixes.keySet())
316 				if (location.startsWith(prefix)
317 						&& (prefix.length() > longestPrefix.length()))
318 					longestPrefix = prefix;
319 
320 			if (longestPrefix.equals(""))
321 				longestPrefix = "default";
322 
323 			// System.out.println("clazz=" + clazz + " level=" + level +
324 			// "longestPrefix=" + longestPrefix);
325 			if (level <= Util.logPrefixes.get(longestPrefix)) {
326 				final String date = Util.dateFormat.format(new Date());
327 				System.out.println(date + " " + location + " "
328 						+ Util.logLevels[level] + ": " + txt);
329 			}
330 		} catch (final ClassNotFoundException e1) {
331 		}
332 	}

354 	public static String stripWhitepace(final String text) {
355 		int s = 0, e = text.length() - 1;
356 
357 		while (s <= e) {
358 			final int c = text.charAt(s);
359 			if (!((c == ' ') || (c == '\t') || (c == '\r') || (c == '\n')))
360 				break;
361 			s++;
362 		}
363 
364 		while (e >= s) {
365 			final int c = text.charAt(e);
366 			if (!((c == ' ') || (c == '\t') || (c == '\r') || (c == '\n')))
367 				break;
368 			e--;
369 		}
370 
371 		return (s <= e) ? text.substring(s, e + 1) : "";
372 	}
