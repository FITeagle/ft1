12  /**
13   * Auto-hiding dialog that notifies the user that a blocking operation is in
14   * progress (usually when talking to the repository).
15   * 
16   * In principle, the usage pattern is: write your blocking code in an anonymous
17   * ProgressJob's run() method (a Runnable that can throw exceptions), then pass
18   * it to this class. It will show an explanatory dialog, then run the blocking
19   * code in a new thread, wait for it to finish, then hide the dialog.
20   * 
21   * Improvements are:
22   * 
23   * * you can create just one dialog at startup, then show/hide it instead of
24   * creating/destroying it every time. * you can avoid showing it if the
25   * operation takes very little (on the order of hundreds of milliseconds) * you
26   * can avoid hiding it and showing it between consecutive requests
27   * 
28   * Except the first one, the rest are transparent.
29   * 
30   * Exceptions thrown in the threaded code are propagated to the method that
31   * passed the ProgressJob instance, but wrapped in a ProgressException.
32   */

79  	/**
80  	 * Code that should be started from a thread, that runs a given blocking job
81  	 * and notifies the UI event loop when the job finished.
82  	 */
83  	class RunAndWait implements Runnable {
84  		private final ProgressJob job;
85  
86  		public RunAndWait(final ProgressJob job) {
87  			this.job = job;
88  		}
89  
90  		public void run() {
91  			final ProgressException[] ex_f = { null };
92  
93  			try {
94  				this.job.run();
95  			} catch (final Exception e) {
96  				ex_f[0] = new ProgressException(e);
97  			} finally {
98  				ProgressDlg.this.display.asyncExec(new Runnable() {
99  					public void run() {
100 						ProgressDlg.this.jobReady = true;
101 						ProgressDlg.this.jobException = ex_f[0];
102 						ProgressDlg.this.display.wake();
103 					}
104 				});
105 			}
106 		}
107 	}
108 
109 	/**
110 	 * Runs runnable in a new thread, and hides the dlg after the thread exits.
111 	 * 
112 	 * @param text
113 	 *            the notification message to show
114 	 * @param runnable
115 	 *            the job to run
116 	 */
117 	public void run(final String text, final ProgressJob runnable)
118 			throws ProgressException {
119 		this.label.setText(text);
120 		this.jobReady = false;
121 		this.jobException = null;
122 
123 		// to avoid showing the dialog for fast operations, only show this after
124 		// a short delay
125 		this.display.timerExec(500, new Runnable() {
126 			public void run() {
127 				if (!ProgressDlg.this.jobReady)
128 					ProgressDlg.this.dlg.setVisible(true);
129 			}
130 		});
131 
132 		new Thread(new RunAndWait(runnable)).start();
133 
134 		while (!(this.dlg.isDisposed() || this.jobReady))
135 			if (!this.display.readAndDispatch())
136 				this.display.sleep();
137 
138 		if (this.dlg.isDisposed())
139 			throw new ProgressException("Dialog closed by user.");
140 
141 		// to avoid repeated show/hide with consecutive operations, only hide
142 		// this after a short delay
143 		this.display.timerExec(200, new Runnable() {
144 			public void run() {
145 				if (ProgressDlg.this.jobReady)
146 					ProgressDlg.this.dlg.setVisible(false);
147 			}
148 		});
149 
150 		if (this.jobException != null)
151 			throw this.jobException;
152 	}
