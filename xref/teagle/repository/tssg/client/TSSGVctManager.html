180 	public void renameIds(String userName, String vctName, Mapping[] mappings) throws RepositoryException
181 	{
182 		TSSGVct vct = getTSSGVct(userName, vctName);
183 		ArrayList<TSSGResourceInstance> del = new ArrayList<TSSGResourceInstance>();
184 		
185 		for (Mapping m : mappings)
186 		{
187 			if (m.designid.equals(m.runtimeid))
188 				continue;
189 			
190 			Util.warn("Renaming: " + m.designid + " --> " + m.runtimeid);
191 					
192 			Util.warn("Testing " + vct.getResourceInstances().length + " ris");
193 			for (TSSGResourceInstance ri : vct.getResourceInstances())
194 			{
195 				Util.warn("Testing config of ri: " + ri.commonName + " - " + ri.getId());
196 				for (TSSGConfiglet cl : ri.getConfigurationData())
197 				{
198 					Util.warn("Testing cl: " + cl.getId() + " - " + cl.getCommonName() + " - " + cl.getType() + " - " + cl.getParamValue());
199 					if (cl.getType().equals("reference") && cl.getParamValue().equals(m.designid))
200 					{
201 						Util.warn("Updating it");
202 						cl.setParamValue(m.runtimeid);
203 						client.updateObject(cl);
204 						
205 					}
206 					else if (cl.getType().equals("reference-array"))
207 					{
208 						String[] ids = cl.getParamValue().split("/");
209 						boolean update = false;
210 						for (int i = 0; i < ids.length; ++i)
211 						{
212 							try
213 							{
214 								String ref = URLDecoder.decode(ids[i], "utf-8");
215 								if (ref.equals(m.designid))
216 								{
217 									ids[i] = URLEncoder.encode(m.runtimeid, "utf-8");
218 									update = true;
219 								}
220 							}
221 							catch (UnsupportedEncodingException e)
222 							{
223 								throw new RepositoryException(e);
224 							}
225 						}
226 						
227 						if (update)
228 						{
229 							String val = "";
230 							for (int i = 0; i < ids.length; ++i)
231 							{
232 								val += ids[i];
233 								if (i < ids.length - 1)
234 									val += "/";
235 							}
236 							cl.setParamValue(val);
237 							client.updateObject(cl);
238 						}
239 					}
240 					else
241 						Util.warn("No update for cl");
242 				}
243 			}
244 			
245 			TSSGResourceInstance oldInstance = getInstanceById(m.designid);
246 			TSSGResourceInstance newInstance = findInstanceById(m.runtimeid);
247 			
248 			if (newInstance == null)
249 			{
250 				Util.warn("No new instance. Setting commonName of " + oldInstance.getId());
251 				oldInstance.commonName = m.runtimeid;
252 				client.updateObject(oldInstance);
253 			}
254 			else
255 			{
256 				Util.warn("Found instance: " + newInstance.getId());
257 				del.add(oldInstance);
258 				vct.replaceInstance(oldInstance, newInstance);
259 			}
260 		}
261 		
262 		if (del.size() != 0)
263 		{
264 			client.updateObject(vct);
265 			for (TSSGResourceInstance oldInstance : del)
266 			{
267 				Util.debug("Considering deletion of old instance: " + oldInstance.commonName + " " + oldInstance.getId());
268 				considerDeletion(oldInstance);
269 			}
270 		}
