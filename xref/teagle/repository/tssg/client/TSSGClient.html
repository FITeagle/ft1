37  	public TSSGClient(ClassLoader loader, URL repoUrl, Object ... aliases) throws IOException 
38  	{
39  		client = Client.create();
40  		
41  		String userinfo = repoUrl.getUserInfo();
42  		
43  		if (userinfo != null)
44  		{
45  			String[] info = userinfo.split(":");
46  			if (info.length > 1)
47  				client.addFilter(new HTTPBasicAuthFilter(info[0], info[1]));
48  		}
49  		webresource = client.resource(repoUrl.toString());
50  		
51  		xs = Util.newXstream();
52  		xs.useAttributeFor(AbstractTSSGEntity.class, "id");
53  		xs.omitField(AbstractTSSGEntity.class, "client");
54  		xs.omitField(AbstractTSSGEntity.class, "initialized");
55  		xs.omitField(AbstractTSSGEntity.class, "isRoot");
56  		xs.omitField(TSSGResource.class, "url");
57  		xs.omitField(TSSGResource.class, "reservations");
58  		xs.omitField(TSSGResource.class, "parsedReservations");
59  		xs.omitField(TSSGResource.class, "realDescription");
60  		xs.registerConverter(new TSSGEntityConverter(xs.getMapper(), xs.getReflectionProvider()));
61  		xs.registerConverter(new ListConverter(xs.getMapper()));
62  		//xs.registerConverter(new RootElementConverter(xs.getMapper(), xs.getReflectionProvider()));
63  		
64  		if (loader!=null) 
65  			xs.setClassLoader(loader);
66  		
67  		mapping = new HashMap<Class<?>, String>();
68  		addAliases(aliases);
69  	}

73  	{
74  		if (aliases.length % 2 != 0)

355 	private String basicMangle(String xml) throws RepositoryException
356 	{
357 		if (xml.equals("<ptmInfo/>"))
358 			return "<ptmInfoInstance/>";
359 		int i = xml.indexOf(">");
360 		int j = xml.lastIndexOf(">");
361 		
362 		if (i < 2 || j < 2 || j <= i)
363 			throw new RepositoryException("Cannot mangle: " + xml);
364 		
365 		if (xml.charAt(i - 1) == '"')
366 		{
367 			int k = xml.indexOf(" ");
368 			if (k >= i)
369 				throw new RepositoryException("Cannot mangle: " + xml);
370 			return xml.substring(0, k) + "Instance" + xml.substring(i, j) + "Instance" + xml.substring(j);
371 		}
372 		else
373 			return xml.substring(0, i) + "Instance" + xml.substring(i, j) + "Instance" + xml.substring(j); 
374 	}
